Example 2

/* parto da un HEX e lo leggo in OKLCH */
#ex-2 {
  background-color: oklch(from #6366F1 l c h);
}

/* poi lo manipolo: stesso colore, solo più scuro */
#ex-2:hover {
  background-color: oklch(from #6366F1 calc(l - 0.2) c h);
}

:root {
  --primary: oklch(0.585 0.204 300); /* purple */

  --l-step: 0.083;

  /* Lightness levels */
  --l-1: calc(2 * var(--l-step));
  --l-2: calc(3 * var(--l-step));
  --l-3: calc(4 * var(--l-step));
  --l-4: calc(5 * var(--l-step));
  --l-5: calc(6 * var(--l-step));
  --l-6: calc(7 * var(--l-step));
  --l-7: calc(8 * var(--l-step));
  --l-8: calc(9 * var(--l-step));
  --l-9: calc(10 * var(--l-step));
}

/* Three palettes from ONE color — only L, C or H changes. */
.scale-primary   { --base: var(--primary); }
.scale-secondary { --base: oklch(from var(--primary) l c calc(h + 180)); } /* hue ruotato di 180° */
.scale-neutral   { --base: oklch(from var(--primary) l 0.01 h); }    /* drop chroma → grey */

.swatch {
  flex: 0 0 64px;
  width: 64px;
  height: 64px;
  background-color: oklch(from var(--base) var(--l) c h);
}

/* position in the row = step on the lightness ramp */
.swatch:nth-child(1) { --l: var(--l-9); }
.swatch:nth-child(2) { --l: var(--l-8); }
.swatch:nth-child(3) { --l: var(--l-7); }
.swatch:nth-child(4) { --l: var(--l-6); }
.swatch:nth-child(5) { --l: var(--l-5); }
.swatch:nth-child(6) { --l: var(--l-4); }
.swatch:nth-child(7) { --l: var(--l-3); }
.swatch:nth-child(8) { --l: var(--l-2); }
.swatch:nth-child(9) { --l: var(--l-1); }
